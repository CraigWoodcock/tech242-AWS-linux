#!/bin/bash

#update
echo -e "\x1b[34mUpdating Lists...\x1b[0m"
echo " "
sudo apt update -y
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

#upgrade
echo -e "\x1b[34mUpgrading Lists...\x1b[0m"
echo " "
sudo DEBIAN_FRONTEND=noninteractive apt upgrade -y
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

# install maven
echo -e "\x1b[34mInstalling Maven...\x1b[0m"
echo " "
sudo DEBIAN_FRONTEND=noninteractive apt install maven -y
echo -e "\x1b[32mMaven Installed...\x1b[0m"
echo " "

#check maven is installed
echo -e "\x1b[34mCheck Version Maven...\x1b[0m"
echo " "
mvn -v
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

#install jdk java 17
echo -e "\x1b[34mInstalling Java 17...\x1b[0m"
echo " "
sudo DEBIAN_FRONTEND=noninteractive apt install openjdk-17-jdk -y
echo -e "\x1b[32mJava Installed...\x1b[0m"
echo " "

#check java is installed
echo -e "\x1b[34mChecking Java Version...\x1b[0m"
echo " "
java -version
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

#Updating Alternatives to use Java 17
echo -e "\x1b[34mUpdating Alternatives...\x1b[0m"
echo " "
sudo update-alternatives --set java /usr/lib/jvm/java-17-openjdk-amd64/bin/java
sudo update-alternatives --set javac /usr/lib/jvm/java-17-openjdk-amd64/bin/javac
echo -e "\x1b[32mJava 17 set\x1b[0m"
echo " "

#install jq
echo -e "\x1b[34mInstalling jq...\x1b[0m"
echo " "
sudo DEBIAN_FRONTEND=noninteractive apt install jq -y
echo -e "\x1b[32mjq Installed...\x1b[0m"
echo " "

#install aws cli v2 (idempotent - skip if already present)
echo -e "\x1b[34mInstalling AWS CLI...\x1b[0m"
echo " "
if command -v aws &> /dev/null; then
    echo -e "\x1b[32mAWS CLI already installed, skipping.\x1b[0m"
else
    sudo DEBIAN_FRONTEND=noninteractive apt install unzip curl -y
    curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "/tmp/awscliv2.zip"
    unzip -o /tmp/awscliv2.zip -d /tmp
    sudo /tmp/aws/install --update
    rm -rf /tmp/awscliv2.zip /tmp/aws
    echo -e "\x1b[32mAWS CLI Installed...\x1b[0m"
fi
aws --version
echo " "

#install MySql server
echo -e "\x1b[34mInstalling MySql Server...\x1b[0m"
echo " "
sudo DEBIAN_FRONTEND=noninteractive apt install mysql-server -y
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

#start enable and check status
echo -e "\x1b[34mStarting Service, Checking Status...\x1b[0m"
echo " "
sudo systemctl is-active --quiet mysql || sudo systemctl start mysql
sudo systemctl is-enabled --quiet mysql || sudo systemctl enable mysql
echo -e "\x1b[32mStarted and Enabled...\x1b[0m"
echo " "

#restart mysql server
echo -e "\x1b[34mRestarting Service...\x1b[0m"
echo " "
sudo systemctl restart mysql
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

#copy app code to this vm
echo -e "\x1b[34mCloning files from git...\x1b[0m"
echo " "
git clone https://github.com/CraigWoodcock/Fishing-App.git repo
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

echo -e "\x1b[34mRenaming application.properties...\x1b[0m"
mv repo/src/main/resources/application.properties.bk repo/src/main/resources/application.properties
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

#fetch secrets from AWS Secrets Manager and set env variables
echo -e "\x1b[34mFetching secrets from Secrets Manager...\x1b[0m"
echo " "

SECRET_NAME="fishing-app/prod"
AWS_REGION="eu-west-1"

SECRET_JSON=$(aws secretsmanager get-secret-value \
    --secret-id "$SECRET_NAME" \
    --region "$AWS_REGION" \
    --query SecretString \
    --output text)

if [ -z "$SECRET_JSON" ]; then
    echo -e "\x1b[31mFailed to retrieve secret. Aborting.\x1b[0m"
    exit 1
fi

export DB_HOST=$(echo "$SECRET_JSON" | jq -r .DB_HOST)
export DB_USER=$(echo "$SECRET_JSON" | jq -r .DB_USER)
export DB_PASS=$(echo "$SECRET_JSON" | jq -r .DB_PASS)
export DB_IP=$(echo "$SECRET_JSON" | jq -r .DB_IP)
export DB_NAME=$(echo "$SECRET_JSON" | jq -r .DB_NAME)
export JWT_SECRET=$(echo "$SECRET_JSON" | jq -r .JWT_SECRET)
export MYSQL_PWD="$DB_PASS"

echo -e "\x1b[32mSecrets loaded successfully...\x1b[0m"
echo " "

#install apache web server
echo -e "\x1b[34mInstalling Apache web Server...\x1b[0m"
echo " "
sudo DEBIAN_FRONTEND=noninteractive apt install apache2 -y
echo -e "\x1b[32mApache Installed...\x1b[0m"
echo " "

#start apache
echo -e "\x1b[34mStarting apache & Enabling apache...\x1b[0m"
echo " "
sudo DEBIAN_FRONTEND=noninteractive systemctl start apache2
sudo DEBIAN_FRONTEND=noninteractive systemctl enable apache2
echo -e "\x1b[32mApache Enabled...\x1b[0m"
echo " "

# Enable Apache modules proxy and http
echo -e "\x1b[34mEnabling Modules...\x1b[0m"
echo " "
sudo a2enmod proxy
sudo a2enmod proxy_http
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

#Configure VirtualHost Config
echo -e "\x1b[34mConfiguring reverse proxy...\x1b[0m"
echo " "
if grep -q 'ProxyPass / http://localhost:8080/' /etc/apache2/sites-available/000-default.conf; then
    echo -e "\x1b[32mReverse proxy already configured.\x1b[0m"
else
    sudo sed -i '/DocumentRoot \/var\/www\/html/ a\ ProxyPreserveHost On\nProxyPass \/ http:\/\/localhost:8080\/\nProxyPassReverse \/ http:\/\/localhost:8080\/\n' /etc/apache2/sites-available/000-default.conf
    echo -e "\x1b[32mReverse proxy configured.\x1b[0m"
fi
echo " "

# Restart Apache
echo -e "\x1b[34mRestarting apache...\x1b[0m"
echo " "
sudo systemctl restart apache2
echo -e "\x1b[32mDone...\x1b[0m"
echo " "

# Check if connection to the database can be established
echo -e "\x1b[34mConnecting to database...\x1b[0m"
echo " "
if mysql -u"$DB_USER" -h"$DB_IP" -e "use $DB_NAME"; then
    echo -e "\x1b[32mConnected to the database. Starting the application...\x1b[0m"
    cd repo
    mvn clean package
    java -jar target/Fishing-App-0.0.1-SNAPSHOT.jar
else
    echo -e "\x1b[31mFailed to connect to the database. Application start aborted.\x1b[0m"
fi
